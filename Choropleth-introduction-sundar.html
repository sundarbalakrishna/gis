# To visualize 'mortality', we set this as a variable for visualization on the map
variable = 'mortality'

# set the range for the choropleth
vmin, vmax = 120, 260

# create figure and axes for Matplotlib
fig, ax = plt.subplots(1, figsize=(10, 6))

# create map
merged.plot(column=variable, cmap='Blues', linewidth=0.8, ax=ax, edgecolor='0.4')

# Since the axis is irrelevant, we remove them:
ax.axis('off')

# add a title
ax.set_title('Mortality or the Death rate in London - 2014')

# create an annotation for the  data source
ax.annotate('Source: London Datastore, 2014',  xy=(0.1, .1), xycoords='figure fraction',
           horizontalalignment='left', verticalalignment='top',
           fontsize=10)
